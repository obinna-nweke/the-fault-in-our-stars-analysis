d = {}
for l in lines:
    l = l.strip()
    if not l:
        continue
    words = l.split()
    caps = [w.rstrip('.’,;?!:\"') for w in words if w[0].isupper()]
    caps = [w.replace("\'s", '') for w in caps]
    caps = [name_mapping.get(w, w) for w in caps if w not in excluded]
    for n in caps:
        if n in d:
            d[n] += 1
        else:
            d[n] = 1

mains = sorted(d.items(), key=lambda x: x[1], reverse=True)
